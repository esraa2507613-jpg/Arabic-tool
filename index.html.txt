document.addEventListener('DOMContentLoaded', function() {

const textarea = document.querySelector('textarea');
const correctedText = document.getElementById('corrected-text');
const tryButton = document.querySelector('.hero .btn');

});

textarea.addEventListener('input', function() {
let text = textarea.value;

let corrected = text
.replace(/اللغه/gi, 'اللغة')
.replace(/إملائيه/gi, 'إملائية')
.replace(/كتابه/gi, 'كتابة')
.replace(/دقيقه/gi, 'دقيقة');

correctedText.textContent = corrected;
});

const addButtons = document.querySelectorAll('.suggestion-item .btn');
addButtons.forEach(button => {
button.addEventListener('click', function() {
alert('تمت إضافة الكلمة البديلة إلى قاموسك الشخصي');
});
});

});
